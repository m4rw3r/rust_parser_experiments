Macro for using monadic do like haskell's do-notation.

#[macro_use]
extern crate parser;
 
fn main() {
    use parser::{Parser, bind, ret, char, any, Error};

    let m = From::from(b"a:b");

    let r: Parser<_, (u8, u8), parser::Error<_>> = mdo!{m,
        first  = any;
                 char(b':');
        second = any;

        ret (first, second)
    };

    assert_eq!(r.unwrap(), (b'a', b'b'));
}
